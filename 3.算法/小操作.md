int is_prime(int x){
    if(x <= 1) return  0;
    for(int i = 2; i * i <= x; i++){
        if(x % i == 0) return 0;
    }
    return 1;
}

int is_palindrome(int x){
    int m = x,n = 0;
    while(m) {
        n = n * 10 + m % 10;
        m /= 10;
    }
    return n == x;
}

    int ans = 1;

        if(b & 1) ans *= a;
        a *= a;
